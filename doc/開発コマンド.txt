環境構築手順

#コンテナ作成
docker-compose build
#reactに必要なパッケージをインストール
docker-compose run --rm frontend sh -c "cd react-project && npm install"
#reactのアプリをビルド
docker compose run --rm frontend sh -c "cd /src/react-project && npm run build"
#コンテナ立ち上げ
docker-compose up -d
